func sum(nums ...int) int {  // ...int - срез из всех переданных чисел
    total := 0

Замыкание (closure)

go
Copy
Download
func counter() func() int {
    count := 0  // переменная "запоминается" внутри возвращаемой функции
    return func() int {
        count++      // захватывает count из внешней функции
        return count
    }
}

Каждый вызов counter() создает свою "замороженную" переменную count.

func task31Handler(w http.ResponseWriter, r *http.Request) {
    // Парсим multipart-форму с лимитом 10MB
    r.ParseMultipartForm(10 << 20)  // 10 * 2^20 = 10MB

Copy

var upgrader = websocket.Upgrader{
    CheckOrigin: func(r *http.Request) bool { return true },  // разрешаем любые источники
}

func wsHandler(w http.ResponseWriter, r *http.Request) {
    // Превращаем HTTP-соединение в WebSocket
    conn, _ := upgrader.Upgrade(w, r, nil)
    defer conn.Close()  // закрыть при выходе
    
    for {
        // Читаем сообщение (messageType, message, error)
        _, msg, err := conn.ReadMessage()
        if err != nil {
            break  // клиент отключился
        }
        // Отправляем обратно
        conn.WriteMessage(websocket.TextMessage, []byte("Echo: "+string(msg)))
    }
}

Важные концепции Go, показанные в коде:

  1. defer - отложенное выполнение

  2. Горутины - легковесные потоки (go func())

  3. Каналы - chan<- и <-chan для коммуникации между горутинами

  4. Интерфейсы - http.ResponseWriter, io.Reader/Writer

  5. Пустой импорт - _ "package" для init-функций

  6. Variadic функции - ...int

  7. Замыкания - функции, захватывающие внешние переменные

  8. Методы с получателями - по значению и по указателю

  9. Несколько возвращаемых значений - (result, error)

  10. Встраивание структур - gorm.Model

Разбор символов:

  • package - ключевое слово, объявляющее пакет (модуль)

  • main - специальное имя: исполняемая программа обязана иметь пакет main

  • Пакет с func main() становится исполняемым файлом (в отличие от библиотечного пакета)

    "bufio"
    "database/sql"

Разбор:

  • import - директива подключения других пакетов

  • ( и ) - группировка множественных импортов (можно и без скобок: import "fmt")

  • Кавычки " - строковый литерал, указывает путь к пакету

  • _ "github.com/microsoft/go-mssqldb" - подчеркивание = псевдоним для пустого импорта:

    • Игнорируем возвращаемое значение (здесь - импортируем только ради функции init() драйвера)

  • <- - оператор канала

  • ... - variadic параметр или распаковка среза

  • range - итерация по коллекциям

  • Множественное возвращение: (result, error)

  1. json.NewDecoder vs json.Unmarshal:

  3. select {} - минимальный способ вечно блокировать горутину без нагрузки на CPU

  4. Проблема в коде: Игнорирование ошибок через _ - так делать нельзя в продакшене!
